# Test file for testing regular expressions, we keep things to lines
# so that we don't need to worry about quotes and delimeters
#
# The format is:
#
#       /<regex>
#       <string>
#       <g0 match>          -- 0: 1,9          (start to end)
#       <g1 match>          -- 1: 4,5          (start to end)
#       <blank>             -- for no more matches
#
/abc
hereabcthere
0: 4,7

/(abc)
hereabcthere
0: 4,7
1: 4,7

/(.*)(abc)(.*)
hereabcthere
0: 0,12
1: 0,4
2: 4,7
3: 7,12

/^one(a+)two
oneaaaatwo
0: 0,10
1: 3,7

/([0-9]{2})-(\w+)
42-hello
0: 0,8
1: 0,2
2: 3,8

/(ab|cd)+ef
abcdabef
0: 0,8
1: 4,6

/^(\w+)\s+(\w+)$
hello world
0: 0,11
1: 0,5
2: 6,11

/([A-Za-z]+)\s+\1
test test
0: 0,9
1: 0,4

/(a*)(b*)(c*)
aaabbc
0: 0,6
1: 0,3
2: 3,5
3: 5,6

/^(.)(.)(.)\3\2\1$
abccba
0: 0,6
1: 0,1
2: 1,2
3: 2,3
